    // List of companies
    const companies = [
      {name: "AWS", type: "Partner", description: "Cloud infrastructure & enterprise solutions", logo: "https://logos-world.net/wp-content/uploads/2021/04/Amazon-AWS-Logo.png", link: "https://aws.amazon.com"},
      {name: "Microsoft Azure", type: "Partner", description: "Cloud computing & enterprise services", logo: "https://logos-world.net/wp-content/uploads/2020/04/Microsoft-Azure-Logo.png", link: "https://azure.microsoft.com"},
      {name: "Google Cloud", type: "Partner", description: "Data, analytics, and AI services", logo: "https://upload.wikimedia.org/wikipedia/commons/5/5f/Google_Cloud_logo.svg", link: "https://cloud.google.com"},
      {name: "Deloitte Digital", type: "Trustee", description: "Digital advisory & risk management", logo: "https://upload.wikimedia.org/wikipedia/commons/7/7b/Deloitte.svg", link: "https://www.deloitte.com"},

    ];

    // Inject slides dynamically
    const sliderWrapper = document.getElementById("company-slider");
    companies.forEach(c => {
      const slide = document.createElement("div");
      slide.className = "swiper-slide";
      slide.innerHTML = `
        <a href="${c.link}" target="_blank">
          <img src="${c.logo}" alt="${c.name}" class="company-logo">
          <div class="company-type">${c.type}</div>
          <div class="company-description">${c.description}</div>
        </a>
      `;
      sliderWrapper.appendChild(slide);
    });

    // Initialize Swiper
    const swiper = new Swiper('.swiper-container', {
      loop: true,
      slidesPerView: 3,
      spaceBetween: 30,
      navigation: {
        nextEl: '.swiper-button-next',
        prevEl: '.swiper-button-prev',
      },
      pagination: {
        el: '.swiper-pagination',
        clickable: true,
      },
      autoplay: {
        delay: 4000,
        disableOnInteraction: false,
      },
      breakpoints: {
        640: { slidesPerView: 1 },
        768: { slidesPerView: 2 },
        1024: { slidesPerView: 3 },
      }
    });
  </script>
